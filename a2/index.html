var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 => используется префиксная форма инкремента, поэтому сразу виден результат
d = b++; alert(d); // 1 => используется постфиксная форма инкремента, поэтому показывается значение переменной до увеличения
c = (2+ ++a); alert(c); // 5 => если разбить выражение на составляющие, то получится: а=а+1, где а = 2, так как в первом действие к а уже прибавили 1, используя префиксная форма инкремента, следовательно а+1=3. с=2+3=5
d = (2+ b++); alert(d); // 4 => если разбить выражение на составляющие, то получится: d=2+b, b= b+1, т.к. используется постфиксная форма инкремента, поэтому применятся будет значение переменной до увеличения, т.е. b= 1+1 =2, и следовательно d=4
alert(a); // 3 в обоих случаях конечное значение равно 3, так как действие инкремента произведено ранее, выводится конечный результат

task 7.
Сравнить null и 0. Попробуйте объяснить результат.
Сравним null с нулём:
alert( null > 0 ); // (1) false
alert( null == 0 ); // (2) false
alert( null >= 0 ); // (3) true
Результат последнего сравнения говорит о том, что "null больше или равно нулю", тогда результат одного из сравнений выше должен быть true, но они оба ложны.
Причина в том, что нестрогое равенство и сравнения > < >= <= работают по-разному. Сравнения преобразуют null в число, рассматривая его как 0. Поэтому выражение (3) null >= 0 истинно, а null > 0 ложно.
С другой стороны, для нестрогого равенства == значений undefined и null действует особое правило: эти значения ни к чему не приводятся, они равны друг другу и не равны ничему другому. Поэтому (2) null == 0 ложно.
